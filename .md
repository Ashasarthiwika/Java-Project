#include <stdio.h>
#define MAX_STUDENTS 100
int main() {
    int marks[MAX_STUDENTS];
    int count = 0;  
    int choice, n, i;
    float average;
    int sum, highest;
    while (1) {
        printf("\n===== Student Management System =====\n");
        printf("1. Add Student Marks\n");
        printf("2. Display All Student Marks\n");
        printf("3. Calculate Average Marks\n");
        printf("4. Find Highest Marks\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);
        switch (choice) {
            case 1:
                printf("Enter number of students to add: ");
                scanf("%d", &n);

                if (count + n > MAX_STUDENTS) {
                    printf("Cannot add more than %d students!\n", MAX_STUDENTS);
                } else {
                    for (i = 0; i < n; i++) {
                        printf("Enter marks for student %d: ", count + 1);
                        scanf("%d", &marks[count]);
                        count++;
                    }
                    printf("Student marks added successfully!\n");
                }
                break;
            case 2:
                if (count == 0) {
                    printf("No student data available!\n");
                } else {
                    printf("\nStudent Marks:\n");
                    for (i = 0; i < count; i++) {
                        printf("Student %d: %d\n", i + 1, marks[i]);
                    }
                }
                break;
            case 3:
                if (count == 0) {
                    printf("No student data available!\n");
                } else {
                    sum = 0;
                    for (i = 0; i < count; i++) {
                        sum += marks[i];
                    }
                    average = (float)sum / count;
                    printf("Average Marks: %.2f\n", average);
                }
                break;
            case 4:
                if (count == 0) {
                    printf("No student data available!\n");
                } else {
                    highest = marks[0];
                    for (i = 1; i < count; i++) {
                        if (marks[i] > highest) {
                            highest = marks[i];
                        }
                    }
                    printf("Highest Marks: %d\n", highest);
                }
                break;
            case 5:
                printf("Exiting program...\n");
                return 0;
            default:
                printf("Invalid choice! Please try again.\n");
        }
    }
    return 0;
}
